package com.ssl.rmas.exceptionhandler;


import java.lang.reflect.Method;
import java.util.UUID;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.aop.interceptor.AsyncUncaughtExceptionHandler;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.messaging.Message;
import org.springframework.security.core.Authentication;
import org.springframework.stereotype.Component;

import com.ssl.rmas.entities.HeaderKeys;
import com.ssl.rmas.managers.ResultsManager;
import org.springframework.beans.factory.annotation.Qualifier;
import com.ssl.rmas.exceptionhandler.ExceptionToMessagesMapper;

@Component
public class ActivityAsyncUncaughtExceptionHandler implements AsyncUncaughtExceptionHandler {

    private static final Logger LOGGER = LoggerFactory.getLogger(ActivityAsyncUncaughtExceptionHandler.class);

    private ExceptionToMessagesMapper exceptionToMessageMapper;
    private ResultsManager resultsManager;

    @Autowired
    @Qualifier("ExceptionToMessagesMapper")
    public void setExceptionToMessageMapper(ExceptionToMessagesMapper exceptionToMessageMapper) {
        this.exceptionToMessageMapper = exceptionToMessageMapper;
    }

    @Autowired
    public void setResultsManager(ResultsManager resultsManager) {
        this.resultsManager = resultsManager;
    }

    @Override
    public void handleUncaughtException(Throwable thrwbl, Method method, Object... os) {
        LOGGER.warn("Async exception while executing {}", method, thrwbl);
        UUID uuid = getUuid(os);
        Authentication currentUser = getCurrentUser(os);
        if (uuid != null) {
            resultsManager.updateResult(currentUser, uuid, exceptionToMessageMapper.getErrorMessages(thrwbl));
        } else {
            LOGGER.error("Could not write an operation result as could not determine activity id");
        }
    }

    private UUID getUuid(Object[] os) {
        for (Object obj : os) {
            if (obj != null && Message.class.isAssignableFrom(obj.getClass())) {
                return ((Message<?>) obj).getHeaders().get(HeaderKeys.ACTIVITY_ID.toString(), UUID.class);
            }
        }
        return null;
    }

    private Authentication getCurrentUser(Object[] os) {
        for (Object obj : os) {
            if (obj != null && Message.class.isAssignableFrom(obj.getClass())) {
                return ((Message<?>) obj).getHeaders().get(HeaderKeys.CURRENT_USER.toString(), Authentication.class);
            }
        }
        return null;
    }
}
