package com.ssl.rmas.endpoints;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.integration.annotation.Transformer;
import org.springframework.integration.support.MessageBuilder;
import org.springframework.messaging.Message;
import org.springframework.security.core.Authentication;
import org.springframework.security.core.context.SecurityContextHolder;

import com.ssl.rmas.entities.HeaderKeys;

public class AddSecurityContextHeaderTransformer {

    private final Logger logger = LoggerFactory.getLogger(AddSecurityContextHeaderTransformer.class);

    @Transformer
    public Message<?> addSecurityContextHeader(Message<?> message) {
        Authentication auth = SecurityContextHolder.getContext().getAuthentication();
        if(auth==null) throw new IllegalStateException("Security context should not be null");
        logger.debug("Adding message header {}: {}", HeaderKeys.CURRENT_USER.toString(), auth);
        return MessageBuilder.fromMessage(message)
                .setHeader(HeaderKeys.CURRENT_USER.toString(), auth)
                .build();
    }
}
